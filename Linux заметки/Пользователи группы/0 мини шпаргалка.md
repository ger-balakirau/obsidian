
```bash
ls -l                    # права
ls -ld dir               # права директории
ls -li                   # inode + права
stat file                # подробности файла

chmod 644 file           # обычный файл
chmod 755 dir            # обычная директория
chmod +x script.sh       # сделать исполняемым

chown user file          # изменить владельца
chown user:group file    # изменить владельца и группу
chgrp group file         # изменить группу

chmod u+s file           # setuid
chmod g+s dir            # setgid
chmod +t dir             # sticky bit

ln file hard             # hard link
ln -s file link          # symlink
readlink link            # цель symlink

umask                    # посмотреть umask
umask 022                # задать umask

getfacl file             # посмотреть ACL
setfacl -m u:user:rw file
setfacl -m g:group:rw file
setfacl -d -m g:group:rwx dir
setfacl -b file          # убрать ACL
```
